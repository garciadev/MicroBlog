CREATE TABLE `category` (
  `categoryID` INTEGER(11) NOT NULL AUTO_INCREMENT,
  `category` VARCHAR(50) COLLATE utf8_general_ci NOT NULL,
  `slug` VARCHAR(50) COLLATE utf8_general_ci NOT NULL,
  `createdDate` DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  `updatedDate` DATETIME DEFAULT NULL,
  PRIMARY KEY USING BTREE (`categoryID`),
  UNIQUE KEY `CategoryID` USING BTREE (`categoryID`)
) ENGINE=InnoDB
AUTO_INCREMENT=4 ROW_FORMAT=DYNAMIC CHARACTER SET 'utf8' COLLATE 'utf8_general_ci'
;


CREATE TABLE `content` (
  `contentID` INTEGER(11) NOT NULL AUTO_INCREMENT,
  `title` VARCHAR(150) COLLATE utf8_general_ci NOT NULL,
  `permaLink` VARCHAR(50) COLLATE utf8_general_ci NOT NULL,
  `body` TEXT COLLATE utf8_general_ci NOT NULL,
  `createdDate` DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  `updatedDate` DATETIME DEFAULT NULL,
  PRIMARY KEY USING BTREE (`contentID`),
  UNIQUE KEY `ContentID` USING BTREE (`contentID`)
) ENGINE=InnoDB
AUTO_INCREMENT=4 ROW_FORMAT=DYNAMIC CHARACTER SET 'utf8' COLLATE 'utf8_general_ci'
;


CREATE TABLE `content_category` (
  `contentID` INTEGER(11) NOT NULL,
  `categoryID` INTEGER(11) NOT NULL
) ENGINE=InnoDB
ROW_FORMAT=DYNAMIC CHARACTER SET 'utf8' COLLATE 'utf8_general_ci'
;


CREATE TABLE `user` (
  `userID` INTEGER(11) NOT NULL AUTO_INCREMENT,
  `firstName` VARCHAR(50) COLLATE utf8_general_ci NOT NULL,
  `lastName` VARCHAR(50) COLLATE utf8_general_ci NOT NULL,
  `email` VARCHAR(150) COLLATE utf8_general_ci NOT NULL,
  `username` VARCHAR(150) COLLATE utf8_general_ci DEFAULT NULL,
  `password` VARCHAR(250) COLLATE utf8_general_ci NOT NULL,
  `createdDate` DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  `updatedDate` DATETIME DEFAULT NULL,
  PRIMARY KEY USING BTREE (`userID`)
) ENGINE=InnoDB
AUTO_INCREMENT=8 ROW_FORMAT=DYNAMIC CHARACTER SET 'utf8' COLLATE 'utf8_general_ci'
;
